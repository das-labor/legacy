\begin{frame}
	\frametitle{Flags}
	\vspace{1cm}

	\textbf{Flags}
	\\
	Meta Informationen "uber den Zustand eines Nutzers.
	\begin{itemize}
		\item \texttt{active}: Benutzeraccount aktiv
		\item \texttt{permanent}: Benutzeraccount darf beliebig oft verwendet werden
		\item \texttt{hnick}: HMAC des Pseudonyms
	\end{itemize}
\end{frame}
